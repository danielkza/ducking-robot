file(GLOB_RECURSE ASSETS RELATIVE "${CMAKE_CURRENT_SOURCE_DIR}" "*")
list(REMOVE_ITEM ASSETS "CMakeLists.txt")

foreach(ASSET ${ASSETS})
    MESSAGE("${CMAKE_RUNTIME_OUTPUT_DIRECTORY}")
    configure_file("${CMAKE_CURRENT_SOURCE_DIR}/${ASSET}" "${CMAKE_RUNTIME_OUTPUT_DIRECTORY}/${ASSET}" COPYONLY)
endforeach(ASSET ASSETS)
